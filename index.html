写在前面

  前面我们手动实现了express中间件机制，这篇文章将讲述如何手动实现koa2中间件的机制。koa2和express框架相比，核心 api 仍然是use方法，不过koanpm 包将路由相关的功能拆分到koa-router包中去了，因此暂时不需要实现express中类似get、post方法了。另外koa2框架原生支持async/await异步机制。

koa2 中间件示例

  koa2 通过app.use()注册中间件函数，app.use()只有 1 个参数，那就是一个中间件处理逻辑的回调函数，函数一般用使用async包裹，这样函数内部可以使用await来获取异步信息。回调函数有两个参数：ctx、next,ctx是将http协议中request和response整合之后的变量，通过它可以拿到http请求和响应的信息。next是指代下一个中间件的方法，在函数中通过await next()的方式执行下一个中间件。这里和express不同的是，因为使用了await执行下一个中间件，因此这条语句下面的代码将延后至下一个中间件执行完成后继续执行，如此往复，一层一层构成的洋葱模型。

const Koa = require("./like-koa2");
const app = new Koa();

app.use(async (ctx, next) => {
  await next();
  const rt = ctx["X-Response-Time"];
  console.log(`${ctx.req.method} ${ctx.req.url} - ${rt}`);
});

// x-response-time
app.use(async (ctx, next) => {
  const start = Date.now();
  await next();
  const ms = Date.now() - start;
  ctx["X-Response-Time"] = `${ms}ms`;
});

// response
app.use(async ctx => {
  ctx.res.end("This is koa2");
});

app.listen(8000);

创建类 koa2 class 类

const http = require("http");

class NewKoa2 {
  constructor() {}

  use() {
    // use方法处理逻辑...
  }

  listen() {
    // listen方法处理逻辑...
  }
}

module.exports = NewKoa2;

完善核心 API 处理逻辑

const http = require("http");

class NewKoa2 {
  constructor() {
    this.middlewareList = [];
  }

  use(fn) {
    this.middlewareList.push(fn); // 注册中间件
    return this; // 返回自己，支持链式调用
  }

  callback() {
    // 默认的服务器回调处理逻辑...
  }

  listen(...args) {
    const server = http.createServer(this.callback()); // 创建http服务对象，传入默认的回调函数
    server.listen(...args); // 将参数传入server.listen()
  }
}

// ...

实现callback函数

  callback方法作为默认的服务器实例回调函数，首先组合中间件列表，将注册的中间件列表处理成一个可以不断执行下去的函数，同时将 http server 默认的req、res参数整合为koa2框架默认的ctx变量。

callback() {
  const fn = compose(this.middlewareList);
  return (req, res) => {
    const ctx = this.createContext(req, res);
    return this.handleRequest(ctx, fn);
  };
}

实现compose函数

  compose方法用来组合间件以及执行顺序的函数，传入实例中注册的所有中间件列表，返回一个接收ctx对象的闭包函数，函数中首先定义一个派发函数dispatch，参数是索引值，找到中间件列表索引对应的中间件函数fn，将调用fn之后的返回值作为 Promise 对象返回。如果异常则返回一个reject,然后直接执行第一个中间件函数。

function compose(middlewareList) {
  return function(ctx) {
    function dispatch(i) {
      const fn = middlewareList[i];
      try {
        return Promise.resolve(fn(ctx, dispatch.bind(null, i + 1))); // 将下一个中间件函数作为next参数传入，实现next机制
      } catch (err) {
        return Promise.reject(err);
      }
    }
    return dispatch(0);
  };
}

实现createContext、handleRequest函数

  createContext函数有两个参数：req、res,返回整合处理好的ctx对象。handleRequest函数只是将处理好的变量、方法执行的一个语义化封装

createContext(req, res) {
  const ctx = {
    req,
    res
  };
  ctx.query = req.query;
  return ctx;
}

handleRequest(ctx, fn) {
  return fn(ctx);
}

最终效果

  下面是完整的实现代码

const http = require("http");

function compose(middlewareList) {
  return function(ctx) {
    function dispatch(i) {
      const fn = middlewareList[i];
      try {
        return Promise.resolve(fn(ctx, dispatch.bind(null, i + 1)));
      } catch (err) {
        return Promise.reject(err);
      }
    }
    return dispatch(0);
  };
}

class NewKoa2 {
  constructor() {
    this.middlewareList = [];
  }

  use(fn) {
    this.middlewareList.push(fn);
    return this;
  }

  createContext(req, res) {
    const ctx = {
      req,
      res
    };
    ctx.query = req.query;
    return ctx;
  }

  handleRequest(ctx, fn) {
    return fn(ctx);
  }

  callback() {
    const fn = compose(this.middlewareList);

    return (req, res) => {
      const ctx = this.createContext(req, res);
      return this.handleRequest(ctx, fn);
    };
  }

  listen(...args) {
    const server = http.createServer(this.callback());
    server.listen(...args);
  }
}

module.exports = NewKoa2;

express 中间件示例

  express 通过app.use()注册中间件函数，app.use()有两个参数，第一个参数是要匹配路由的路径字符串，如果不传，默认值为/，匹配所有路由。第二个参数是中间件处理函数，有三个参数req、res、next，分别是 http 请求request对象、response对象、以及指代下一个中间件函数的next方法。当前中间件函数执行完后，如果需要执行下一个中间件函数，只需要调用next方法

const express = require("express"); // 引入express
const app = express(); // 新建express实例

// 处理任意路由
app.use((req, res, next) => {
  console.log("请求开始...", req.method, req.url);
  next();
});

// 处理api路由
app.use("/api", (req, res, next) => {
  next();
});

// 处理 GET api/get路由...
app.get("/api/get", (req, res, next) => {
  res.json({
    code: 0,
    data: "get data"
  });
  next();
});

// 模拟登录检验中间件
const loginCheck = (req, res, next) => {
  setTimeout(() => {
    next();
  }, 1000);
};

// 处理 GET api/post路由...
app.post("/api/post", loginCheck, (req, res, next) => {
  res.json({
    code: 0,
    data: "post data"
  });
  next();
});

// 处理404
app.use((req, res, next) => {
  next();
});

// 监听 3000 端口
app.listen(3000, () => {
  console.log("listen port on 3000...");
});

实现register函数

  register方法用来统一处理中间件参数，如果第一个参数是字符串，则认为是需要匹配的路由路径，将第二个参数以后的所有参数转成数组存入 stack 属性；如果第一个参数不是字符串类型，则默认匹配所有路由，将 path 属性设置为/，并将所有参数转换为数组存入 stack 属性：

const slice = Array.prototype.slice

register(path) {
  const info = {};
  if (typeof path === "string") {
    info.path = path;
    // 从第二个参数开始, 转换为数组，存入stack
    info.stack = slice.call(arguments, 1);
  } else {
    info.path = "/";
    // 从第1个参数开始, 转换为数组，存入stack
    info.stack = slice.call(arguments, 0);
  }
  return info;
}

实现callback函数

  callback方法作为默认的服务器实例回调函数，用来设置 express 特有的一些方法，如res.json等。同时根据req.method的不同筛选出匹配的中间件列表，最后通过 handle 函数去执行中间件。

callback() {
  return (req, res) => {
    res.json = data => {
      res.setHeader("Content-type", "application/json");
      res.end(JSON.stringify(data));
    };
    const url = req.url;
    const method = req.method.toLowerCase();
    const resultList = this.match(method, url);
    this.handle(req, res, resultList);
  };
}

实现match、handle函数

  match函数有两个参数，一个是请求的方法类型，一个是请求的路由路径。根据请求方法不同，先将this.routes.all中和对应方法的routes数组合并，然后遍历生成需要执行的中间件函数列表。handle函数是实现next机制的核心方法，函数内部创建一个next函数，这个函数会取得中间列表第一个函数，如果存在，就会一直执行下去

match(method, url) {
  let stack = [];
  if (url === "/favicon.icon") {
    return stack;
  }
  // 获取 routes
  let curRoutes = [];
  curRoutes = curRoutes.concat(this.routes.all);
  curRoutes = curRoutes.concat(this.routes[method]);
  curRoutes.forEach(routeInfo => {
    if (url.indexOf(routeInfo.path) === 0) {
      stack = stack.concat(routeInfo.stack);
    }
  });
  return stack;
}

handle(req, res, stack) {
  const next = () => {
    // 拿到第一个匹配的中间件
    const middleware = stack.shift();
    if (middleware) {
      // 执行中间件函数
      middleware(req, res, next);
    }
  };
  next();
}

前言

Git 自带有 git diff 命令，能够查看全部/指定文件的差异,由于差异信息过于简单，也缺乏许多 Git 插件提供的增强性功能，一般只用来确认下文件改动的大致内容。

实际开发中习惯使用 VS Code 编辑器，它的插件库中也有着完善丰富的 Git 插件，最初使用的是 GitLens。使用该插件在遇到合并冲突时，有冲突的本地文件会以两种色彩高亮显示本地版本的改动和远程版本的改动，这时的本地文件是两路合并之后的融合版本。

大多数情况下，GitLens 还是能够满足开发需求的。但是，如果开发项目中没有使用统一风格的代码格式检测插件，如 ESLint。在代码格式化风格变化较大的时候，GitLens 插件自动生成的合并代码会出现毫无规律的重复嵌套，使得修复冲突变的复杂且容易出错。这个时候，GitLens 就不如专门的图形界面工具好用了。
